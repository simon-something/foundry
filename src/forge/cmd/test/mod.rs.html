222    /// Returns sources which include any tests to be executed.
223    /// If no filters are provided, sources are filtered by existence of test/invariant methods in
224    /// them, If filters are provided, sources are additionally filtered by them.
225    pub fn get_sources_to_compile(
226        &self,
227        config: &Config,
228        filter: &ProjectPathsAwareFilter,
229    ) -> Result<BTreeSet<PathBuf>> {
230        let mut project = config.create_project(true, true)?;
231        project.update_output_selection(|selection| {
232            *selection = OutputSelection::common_output_selection(["abi".to_string()]);
233        });
234
235        let output = project.compile()?;
236
237        if output.has_compiler_errors() {
238            sh_println!("{output}")?;
239            eyre::bail!("Compilation failed");
240        }
241
242        // ABIs of all sources
243        let abis = output
244            .into_artifacts()
245            .filter_map(|(id, artifact)| artifact.abi.map(|abi| (id, abi)))
246            .collect::<BTreeMap<_, _>>();
247
248        // Filter sources by their abis and contract names.
249        let mut test_sources = abis
250            .iter()
251            .filter(|(id, abi)| matches_contract(id, abi, filter))
252            .map(|(id, _)| id.source.clone())
253            .collect::<BTreeSet<_>>();
254
255        if test_sources.is_empty() {
256            if filter.is_empty() {
257                sh_println!(
258                    "No tests found in project! \
259                        Forge looks for functions that starts with `test`."
260                )?;
261            } else {
262                sh_println!("No tests match the provided pattern:")?;
263                sh_print!("{filter}")?;
264
265                // Try to suggest a test when there's no match
266                if let Some(test_pattern) = &filter.args().test_pattern {
267                    let test_name = test_pattern.as_str();
268                    let candidates = abis
269                        .into_iter()
270                        .filter(|(id, _)| {
271                            filter.matches_path(&id.source) && filter.matches_contract(&id.name)
272                        })
273                        .flat_map(|(_, abi)| abi.functions.into_keys())
274                        .collect::<Vec<_>>();
275                    if let Some(suggestion) = utils::did_you_mean(test_name, candidates).pop() {
276                        sh_println!("\nDid you mean `{suggestion}`?")?;
277                    }
278                }
279            }
280
281            eyre::bail!("No tests to run");
282        }
283
284        // Always recompile all sources to ensure that `getCode` cheatcode can use any artifact.
285        test_sources.extend(source_files_iter(
286            &project.paths.sources,
287            MultiCompilerLanguage::FILE_EXTENSIONS,
288        ));
289
290        Ok(test_sources)
291    }

293    /// Executes all the tests in the project.
294    ///
295    /// This will trigger the build process first. On success all test contracts that match the
296    /// configured filter will be executed
297    ///
298    /// Returns the test results for all matching tests.
299    pub async fn execute_tests(mut self) -> Result<TestOutcome> {
300        // Merge all configs.
301        let (mut config, mut evm_opts) = self.load_config_and_evm_opts()?;
302
303        // Explicitly enable isolation for gas reports for more correct gas accounting.
304        if self.gas_report {
305            evm_opts.isolate = true;
306        } else {
307            // Do not collect gas report traces if gas report is not enabled.
308            config.fuzz.gas_report_samples = 0;
309            config.invariant.gas_report_samples = 0;
310        }
311
312        let should_mutate = self.mutate.is_some();
313
314        // Force dyn test linking for mutation testing
315        if should_mutate {
316            config.dynamic_test_linking = true;
317            config.cache = true;
318        }
319
320        // Install missing dependencies.
321        if install::install_missing_dependencies(&mut config) && config.auto_detect_remappings {
322            // need to re-configure here to also catch additional remappings
323            config = self.load_config()?;
324        }
325
326        // Set up the project.
327        let project = config.project()?;
328
329        let filter = self.filter(&config)?;
330        trace!(target: "forge::test", ?filter, "using filter");
331
332        let sources_to_compile = self.get_sources_to_compile(&config, &filter)?;
333
334        let compiler = ProjectCompiler::new()
335            .dynamic_test_linking(config.dynamic_test_linking)
336            .quiet(shell::is_json() || self.junit)
337            .files(sources_to_compile);
338
339        let output = compiler.compile(&project)?;
340
341        // Create test options from general project settings and compiler output.
342        let project_root = &project.paths.root;
343
344        let should_debug = self.debug;
345        let should_draw = self.flamegraph || self.flamechart;
346
347        // Determine print verbosity and executor verbosity.
348        let verbosity = evm_opts.verbosity;
349        if (self.gas_report && evm_opts.verbosity < 3) || self.flamegraph || self.flamechart {
350            evm_opts.verbosity = 3;
351        }
352
353        let env = evm_opts.evm_env().await?;
354
355        // Enable internal tracing for more informative flamegraph.
356        if should_draw && !self.decode_internal {
357            self.decode_internal = true;
358        }
359
360        // Choose the internal function tracing mode, if --decode-internal is provided.
361        let decode_internal = if self.decode_internal {
362            // If more than one function matched, we enable simple tracing.
363            // If only one function matched, we enable full tracing. This is done in `run_tests`.
364            InternalTraceMode::Simple
365        } else {
366            InternalTraceMode::None
367        };
368
369        // Prepare the test builder.
370        let config = Arc::new(config);
371        let runner = MultiContractRunnerBuilder::new(config.clone())
372            .set_debug(should_debug)
373            .set_decode_internal(decode_internal)
374            .initial_balance(evm_opts.initial_balance)
375            .evm_spec(config.evm_spec_id())
376            .sender(evm_opts.sender)
377            .with_fork(evm_opts.get_fork(&config, env.clone()))
378            .enable_isolation(evm_opts.isolate)
379            .odyssey(evm_opts.odyssey)
380            .build::<MultiCompiler>(project_root, &output, env.clone(), evm_opts.clone())?;
381
382        let libraries = runner.libraries.clone();
383        let mut outcome =
384            self.run_tests(runner, config.clone(), verbosity, &filter, &output).await?;
385
386        if should_draw {
387            let (suite_name, test_name, mut test_result) =
388                outcome.remove_first().ok_or_eyre("no tests were executed")?;
389
390            let (_, arena) = test_result
391                .traces
392                .iter_mut()
393                .find(|(kind, _)| *kind == TraceKind::Execution)
394                .unwrap();
395
396            // Decode traces.
397            let decoder = outcome.last_run_decoder.as_ref().unwrap();
398            decode_trace_arena(arena, decoder).await;
399            let mut fst = folded_stack_trace::build(arena);
400
401            let label = if self.flamegraph { "flamegraph" } else { "flamechart" };
402            let contract = suite_name.split(':').next_back().unwrap();
403            let test_name = test_name.trim_end_matches("()");
404            let file_name = format!("cache/{label}_{contract}_{test_name}.svg");
405            let file = std::fs::File::create(&file_name).wrap_err("failed to create file")?;
406            let file = std::io::BufWriter::new(file);
407
408            let mut options = inferno::flamegraph::Options::default();
409            options.title = format!("{label} {contract}::{test_name}");
410            options.count_name = "gas".to_string();
411            if self.flamechart {
412                options.flame_chart = true;
413                fst.reverse();
414            }
415
416            // Generate SVG.
417            inferno::flamegraph::from_lines(&mut options, fst.iter().map(String::as_str), file)
418                .wrap_err("failed to write svg")?;
419            sh_println!("Saved to {file_name}")?;
420
421            // Open SVG in default program.
422            if let Err(e) = opener::open(&file_name) {
423                sh_err!("Failed to open {file_name}; please open it manually: {e}")?;
424            }
425        }
426
427        if should_debug {
428            // Get first non-empty suite result. We will have only one such entry.
429            let (_, _, test_result) =
430                outcome.remove_first().ok_or_eyre("no tests were executed")?;
431
432            let sources =
433                ContractSources::from_project_output(&output, project.root(), Some(&libraries))?;
434
435            // Run the debugger.
436            let mut builder = Debugger::builder()
437                .traces(
438                    test_result.traces.iter().filter(|(t, _)| t.is_execution()).cloned().collect(),
439                )
440                .sources(sources)
441                .breakpoints(test_result.breakpoints.clone());
442
443            if let Some(decoder) = &outcome.last_run_decoder {
444                builder = builder.decoder(decoder);
445            }
446
447            let mut debugger = builder.build();
448            if let Some(dump_path) = self.dump.clone() {
449                debugger.dump_to_file(&dump_path)?;
450            } else {
451                debugger.try_run_tui()?;
452            }
453        }
454
455        // All test have been run once before reaching this point
456        if should_mutate {
457            // check outcome here, stop if any test failed
458            // @todo rather set non-allowed failed tests in config and ensure_ok() here?
459            // @todo other checks: no fork (or just exclude based on clap arg?)
460            if outcome.failed() > 0 {
461                eyre::bail!("Cannot run mutation testing with failed tests");
462            }
463
464            let mutate_paths = if let Some(pattern) = &self.mutate_path {
465                // If --mutate-path is provided, use it to filter paths
466                source_files_iter(&project.paths.sources, MultiCompilerLanguage::FILE_EXTENSIONS)
467                    .filter(|entry| {
468                        // @todo filter out interfaces here?
469                        // we do it in lexing for now
470                        entry.is_sol() && !entry.is_sol_test() && pattern.is_match(entry)
471                    })
472                    .collect()
473            } else if let Some(contract_pattern) = &self.mutate_contract {
474                // If --mutate-contract is provided, use it to filter contracts
475                source_files_iter(&project.paths.sources, MultiCompilerLanguage::FILE_EXTENSIONS)
476                    .filter(|entry| {
477                        entry.is_sol()
478                            && !entry.is_sol_test()
479                            && output
480                                .artifact_ids()
481                                .find(|(id, _)| id.source == *entry)
482                                .is_some_and(|(id, _)| contract_pattern.is_match(&id.name))
483                    })
484                    .collect()
485            } else if self.mutate.as_ref().unwrap().is_empty() {
486                // If --mutate is passed without arguments, use all Solidity files
487                source_files_iter(&project.paths.sources, MultiCompilerLanguage::FILE_EXTENSIONS)
488                    .filter(|entry| entry.is_sol() && !entry.is_sol_test())
489                    .collect()
490            } else {
491                // If --mutate is passed with arguments, use those paths
492                self.mutate.as_ref().unwrap().clone()
493            };
494
495            sh_println!("Running mutation tests...").unwrap();
496            let mut mutation_summary = MutationsSummary::new();
497
498            for path in mutate_paths {
499                sh_println!("Running mutation tests for {}", path.display()).unwrap();
500
501                // Check if this file has already been tested and if the build id is the
502                // same - if so, just add the mutants to the summary
503                let mut handler = MutationHandler::new(path.clone(), config.clone());
504
505                handler.read_source_contract()?;
506
507                let build_id = output
508                    .artifact_ids()
509                    .find_map(
510                        |(id, _)| if id.source == path { Some(id.build_id.clone()) } else { None },
511                    )
512                    .unwrap_or_default();
513
514                // If we have cached results for these mutations and build id, use them and skip
515                // running tests
516                if let Some(prior) = handler.retrieve_cached_mutant_results(&build_id) {
517                    for (mutant, status) in prior {
518                        match status {
519                            crate::mutation::mutant::MutationResult::Dead => {
520                                mutation_summary.add_dead_mutant(mutant)
521                            }
522                            crate::mutation::mutant::MutationResult::Alive => {
523                                mutation_summary.add_survived_mutant(mutant)
524                            }
525                            crate::mutation::mutant::MutationResult::Invalid => {
526                                mutation_summary.update_invalid_mutant(mutant)
527                            }
528                        }
529                    }
530                    continue;
531                }
532
533                // Try cached mutants first
534                let mut mutants = if let Some(ms) = handler.retrieve_cached_mutants(&build_id) {
535                    ms
536                } else {
537                    // No cache match: generate fresh mutants
538                    handler.generate_ast().await;
539                    handler.mutations.clone()
540                };
541
542                // Accumulate per-mutant results for persistence
543                let mut results_vec: Vec<(
544                    crate::mutation::mutant::Mutant,
545                    crate::mutation::mutant::MutationResult,
546                )> = Vec::with_capacity(mutants.len());
547
548                for (i, mutant) in mutants.iter().enumerate() {
549                    sh_println!("Testing mutant {} out of {}", i + 1, mutants.len()).unwrap();
550
551                    handler.generate_mutated_solidity(&mutant);
552                    let new_filter = self.filter(&config).unwrap();
553                    let compiler = ProjectCompiler::new()
554                        .dynamic_test_linking(config.dynamic_test_linking)
555                        .quiet(true);
556
557                    let compile_output = compiler.compile(&project);
558
559                    if compile_output.is_err() {
560                        mutation_summary.update_invalid_mutant(mutant.clone());
561                        results_vec.push((
562                            mutant.clone(),
563                            crate::mutation::mutant::MutationResult::Invalid,
564                        ));
565                    } else {
566                        let mut runner = MultiContractRunnerBuilder::new(config.clone())
567                            .set_debug(false)
568                            .initial_balance(evm_opts.initial_balance)
569                            .evm_spec(config.evm_spec_id())
570                            .sender(evm_opts.sender)
571                            .odyssey(evm_opts.odyssey)
572                            .build::<MultiCompiler>(
573                                &config.root,
574                                &compile_output.unwrap(),
575                                env.clone(),
576                                evm_opts.clone(),
577                            )?;
578
579                        let results: BTreeMap<String, SuiteResult> =
580                            runner.test_collect(&new_filter)?;
581
582                        let outcome = TestOutcome::new(results, self.allow_failure);
583                        if outcome.failures().count() > 0 {
584                            mutation_summary.add_dead_mutant(mutant.clone());
585                            results_vec.push((
586                                mutant.clone(),
587                                crate::mutation::mutant::MutationResult::Dead,
588                            ));
589                        } else {
590                            mutation_summary.add_survived_mutant(mutant.clone());
591                            results_vec.push((
592                                mutant.clone(),
593                                crate::mutation::mutant::MutationResult::Alive,
594                            ));
595                        }
596                    }
597                }
598
599                handler.restore_original_source();
600
601                // If we generated fresh mutants, persist them for this build id
602                if handler.mutations.len() > 0 && !build_id.is_empty() {
603                    let _ = handler.persist_cached_mutants(&build_id, &handler.mutations);
604                    let _ = handler.persist_cached_results(&build_id, &results_vec);
605                }
606            }
607
608            MutationReporter::new().report(&mutation_summary);
609
610            outcome = TestOutcome::empty(true);
611        }
612
613        Ok(outcome)
614    }

616    /// Run all tests that matches the filter predicate from a test runner
617    pub async fn run_tests(
618        &self,
619        mut runner: MultiContractRunner,
620        config: Arc<Config>,
621        verbosity: u8,
622        filter: &ProjectPathsAwareFilter,
623        output: &ProjectCompileOutput,
624    ) -> eyre::Result<TestOutcome> {
625        if self.list {
626            return list(runner, filter);
627        }
628
629        trace!(target: "forge::test", "running all tests");
630
631        // If we need to render to a serialized format, we should not print anything else to stdout.
632        let silent = self.gas_report && shell::is_json() || self.summary && shell::is_json();
633
634        let num_filtered = runner.matching_test_functions(filter).count();
635        if num_filtered != 1 && (self.debug || self.flamegraph || self.flamechart) {
636            let action = if self.flamegraph {
637                "generate a flamegraph"
638            } else if self.flamechart {
639                "generate a flamechart"
640            } else {
641                "run the debugger"
642            };
643            let filter = if filter.is_empty() {
644                String::new()
645            } else {
646                format!("\n\nFilter used:\n{filter}")
647            };
648            eyre::bail!(
649                "{num_filtered} tests matched your criteria, but exactly 1 test must match in order to {action}.\n\n\
650                 Use --match-contract and --match-path to further limit the search.{filter}",
651            );
652        }
653
654        // If exactly one test matched, we enable full tracing.
655        if num_filtered == 1 && self.decode_internal {
656            runner.decode_internal = InternalTraceMode::Full;
657        }
658
659        // Run tests in a non-streaming fashion and collect results for serialization.
660        if !self.gas_report && !self.summary && shell::is_json() {
661            let mut results = runner.test_collect(filter)?;
662            results.values_mut().for_each(|suite_result| {
663                for test_result in suite_result.test_results.values_mut() {
664                    if verbosity >= 2 {
665                        // Decode logs at level 2 and above.
666                        test_result.decoded_logs = decode_console_logs(&test_result.logs);
667                    } else {
668                        // Empty logs for non verbose runs.
669                        test_result.logs = vec![];
670                    }
671                }
672            });
673            sh_println!("{}", serde_json::to_string(&results)?)?;
674            return Ok(TestOutcome::new(results, self.allow_failure));
675        }
676
677        if self.junit {
678            let results = runner.test_collect(filter)?;
679            sh_println!("{}", junit_xml_report(&results, verbosity).to_string()?)?;
680            return Ok(TestOutcome::new(results, self.allow_failure));
681        }
682
683        let remote_chain_id = runner.evm_opts.get_remote_chain_id().await;
684        let known_contracts = runner.known_contracts.clone();
685
686        let libraries = runner.libraries.clone();
687
688        // Run tests in a streaming fashion.
689        let (tx, rx) = channel::<(String, SuiteResult)>();
690        let timer = Instant::now();
691        let show_progress = config.show_progress;
692        let handle = tokio::task::spawn_blocking({
693            let filter = filter.clone();
694            move || runner.test(&filter, tx, show_progress)
695        });
696
697        // Set up trace identifiers.
698        let mut identifier = TraceIdentifiers::new().with_local(&known_contracts);
699
700        // Avoid using etherscan for gas report as we decode more traces and this will be
701        // expensive.
702        if !self.gas_report {
703            identifier = identifier.with_etherscan(&config, remote_chain_id)?;
704        }
705
706        // Build the trace decoder.
707        let mut builder = CallTraceDecoderBuilder::new()
708            .with_known_contracts(&known_contracts)
709            .with_verbosity(verbosity);
710        // Signatures are of no value for gas reports.
711        if !self.gas_report {
712            builder =
713                builder.with_signature_identifier(SignaturesIdentifier::from_config(&config)?);
714        }
715
716        if self.decode_internal {
717            let sources =
718                ContractSources::from_project_output(output, &config.root, Some(&libraries))?;
719            builder = builder.with_debug_identifier(DebugTraceIdentifier::new(sources));
720        }
721        let mut decoder = builder.build();
722
723        let mut gas_report = self.gas_report.then(|| {
724            GasReport::new(
725                config.gas_reports.clone(),
726                config.gas_reports_ignore.clone(),
727                config.gas_reports_include_tests,
728            )
729        });
730
731        let mut gas_snapshots = BTreeMap::<String, BTreeMap<String, String>>::new();
732
733        let mut outcome = TestOutcome::empty(self.allow_failure);
734
735        let mut any_test_failed = false;
736        for (contract_name, suite_result) in rx {
737            let tests = &suite_result.test_results;
738
739            // Clear the addresses and labels from previous test.
740            decoder.clear_addresses();
741
742            // We identify addresses if we're going to print *any* trace or gas report.
743            let identify_addresses = verbosity >= 3
744                || self.gas_report
745                || self.debug
746                || self.flamegraph
747                || self.flamechart;
748
749            // Print suite header.
750            if !silent {
751                sh_println!()?;
752                for warning in &suite_result.warnings {
753                    sh_warn!("{warning}")?;
754                }
755                if !tests.is_empty() {
756                    let len = tests.len();
757                    let tests = if len > 1 { "tests" } else { "test" };
758                    sh_println!("Ran {len} {tests} for {contract_name}")?;
759                }
760            }
761
762            // Process individual test results, printing logs and traces when necessary.
763            for (name, result) in tests {
764                let show_traces =
765                    !self.suppress_successful_traces || result.status == TestStatus::Failure;
766                if !silent {
767                    sh_println!("{}", result.short_result(name))?;
768
769                    // Display invariant metrics if invariant kind.
770                    if let TestKind::Invariant { metrics, .. } = &result.kind
771                        && !metrics.is_empty()
772                    {
773                        let _ = sh_println!("\n{}\n", format_invariant_metrics_table(metrics));
774                    }
775
776                    // We only display logs at level 2 and above
777                    if verbosity >= 2 && show_traces {
778                        // We only decode logs from Hardhat and DS-style console events
779                        let console_logs = decode_console_logs(&result.logs);
780                        if !console_logs.is_empty() {
781                            sh_println!("Logs:")?;
782                            for log in console_logs {
783                                sh_println!("  {log}")?;
784                            }
785                            sh_println!()?;
786                        }
787                    }
788                }
789
790                // We shouldn't break out of the outer loop directly here so that we finish
791                // processing the remaining tests and print the suite summary.
792                any_test_failed |= result.status == TestStatus::Failure;
793
794                // Clear the addresses and labels from previous runs.
795                decoder.clear_addresses();
796                decoder
797                    .labels
798                    .extend(result.labeled_addresses.iter().map(|(k, v)| (*k, v.clone())));
799
800                // Identify addresses and decode traces.
801                let mut decoded_traces = Vec::with_capacity(result.traces.len());
802                for (kind, arena) in &mut result.traces.clone() {
803                    if identify_addresses {
804                        decoder.identify(arena, &mut identifier);
805                    }
806
807                    // verbosity:
808                    // - 0..3: nothing
809                    // - 3: only display traces for failed tests
810                    // - 4: also display the setup trace for failed tests
811                    // - 5..: display all traces for all tests, including storage changes
812                    let should_include = match kind {
813                        TraceKind::Execution => {
814                            (verbosity == 3 && result.status.is_failure()) || verbosity >= 4
815                        }
816                        TraceKind::Setup => {
817                            (verbosity == 4 && result.status.is_failure()) || verbosity >= 5
818                        }
819                        TraceKind::Deployment => false,
820                    };
821
822                    if should_include {
823                        decode_trace_arena(arena, &decoder).await;
824                        decoded_traces.push(render_trace_arena_inner(arena, false, verbosity > 4));
825                    }
826                }
827
828                if !silent && show_traces && !decoded_traces.is_empty() {
829                    sh_println!("Traces:")?;
830                    for trace in &decoded_traces {
831                        sh_println!("{trace}")?;
832                    }
833                }
834
835                if let Some(gas_report) = &mut gas_report {
836                    gas_report.analyze(result.traces.iter().map(|(_, a)| &a.arena), &decoder).await;
837
838                    for trace in &result.gas_report_traces {
839                        decoder.clear_addresses();
840
841                        // Re-execute setup and deployment traces to collect identities created in
842                        // setUp and constructor.
843                        for (kind, arena) in &result.traces {
844                            if !matches!(kind, TraceKind::Execution) {
845                                decoder.identify(arena, &mut identifier);
846                            }
847                        }
848
849                        for arena in trace {
850                            decoder.identify(arena, &mut identifier);
851                            gas_report.analyze([arena], &decoder).await;
852                        }
853                    }
854                }
855
856                // Collect and merge gas snapshots.
857                for (group, new_snapshots) in &result.gas_snapshots {
858                    gas_snapshots.entry(group.clone()).or_default().extend(new_snapshots.clone());
859                }
860            }
861
862            // Write gas snapshots to disk if any were collected.
863            if !gas_snapshots.is_empty() {
864                // By default `gas_snapshot_check` is set to `false` in the config.
865                //
866                // The user can either:
867                // - Set `FORGE_SNAPSHOT_CHECK=true` in the environment.
868                // - Pass `--gas-snapshot-check=true` as a CLI argument.
869                // - Set `gas_snapshot_check = true` in the config.
870                //
871                // If the user passes `--gas-snapshot-check=<bool>` then it will override the config
872                // and the environment variable, disabling the check if `false` is passed.
873                //
874                // Exiting early with code 1 if differences are found.
875                if self.gas_snapshot_check.unwrap_or(config.gas_snapshot_check) {
876                    let differences_found = gas_snapshots.clone().into_iter().fold(
877                        false,
878                        |mut found, (group, snapshots)| {
879                            // If the snapshot file doesn't exist, we can't compare so we skip.
880                            if !&config.snapshots.join(format!("{group}.json")).exists() {
881                                return false;
882                            }
883
884                            let previous_snapshots: BTreeMap<String, String> =
885                                fs::read_json_file(&config.snapshots.join(format!("{group}.json")))
886                                    .expect("Failed to read snapshots from disk");
887
888                            let diff: BTreeMap<_, _> = snapshots
889                                .iter()
890                                .filter_map(|(k, v)| {
891                                    previous_snapshots.get(k).and_then(|previous_snapshot| {
892                                        if previous_snapshot != v {
893                                            Some((
894                                                k.clone(),
895                                                (previous_snapshot.clone(), v.clone()),
896                                            ))
897                                        } else {
898                                            None
899                                        }
900                                    })
901                                })
902                                .collect();
903
904                            if !diff.is_empty() {
905                                let _ = sh_eprintln!(
906                                    "{}",
907                                    format!("\n[{group}] Failed to match snapshots:").red().bold()
908                                );
909
910                                for (key, (previous_snapshot, snapshot)) in &diff {
911                                    let _ = sh_eprintln!(
912                                        "{}",
913                                        format!("- [{key}] {previous_snapshot}  {snapshot}").red()
914                                    );
915                                }
916
917                                found = true;
918                            }
919
920                            found
921                        },
922                    );
923
924                    if differences_found {
925                        sh_eprintln!()?;
926                        eyre::bail!("Snapshots differ from previous run");
927                    }
928                }
929
930                // By default `gas_snapshot_emit` is set to `true` in the config.
931                //
932                // The user can either:
933                // - Set `FORGE_SNAPSHOT_EMIT=false` in the environment.
934                // - Pass `--gas-snapshot-emit=false` as a CLI argument.
935                // - Set `gas_snapshot_emit = false` in the config.
936                //
937                // If the user passes `--gas-snapshot-emit=<bool>` then it will override the config
938                // and the environment variable, enabling the check if `true` is passed.
939                if self.gas_snapshot_emit.unwrap_or(config.gas_snapshot_emit) {
940                    // Create `snapshots` directory if it doesn't exist.
941                    fs::create_dir_all(&config.snapshots)?;
942
943                    // Write gas snapshots to disk per group.
944                    gas_snapshots.clone().into_iter().for_each(|(group, snapshots)| {
945                        fs::write_pretty_json_file(
946                            &config.snapshots.join(format!("{group}.json")),
947                            &snapshots,
948                        )
949                        .expect("Failed to write gas snapshots to disk");
950                    });
951                }
952            }
953
954            // Print suite summary.
955            if !silent {
956                sh_println!("{}", suite_result.summary())?;
957            }
958
959            // Add the suite result to the outcome.
960            outcome.results.insert(contract_name, suite_result);
961
962            // Stop processing the remaining suites if any test failed and `fail_fast` is set.
963            if self.fail_fast && any_test_failed {
964                break;
965            }
966        }
967        outcome.last_run_decoder = Some(decoder);
968        let duration = timer.elapsed();
969
970        trace!(target: "forge::test", len=outcome.results.len(), %any_test_failed, "done with results");
971
972        if let Some(gas_report) = gas_report {
973            let finalized = gas_report.finalize();
974            sh_println!("{}", &finalized)?;
975            outcome.gas_report = Some(finalized);
976        }
977
978        if !self.summary && !shell::is_json() {
979            sh_println!("{}", outcome.summary(duration))?;
980        }
981
982        if self.summary && !outcome.results.is_empty() {
983            let summary_report = TestSummaryReport::new(self.detailed, outcome.clone());
984            sh_println!("{}", &summary_report)?;
985        }
986
987        // Reattach the task.
988        if let Err(e) = handle.await {
989            match e.try_into_panic() {
990                Ok(payload) => std::panic::resume_unwind(payload),
991                Err(e) => return Err(e.into()),
992            }
993        }
994
995        // Persist test run failures to enable replaying.
996        persist_run_failures(&config, &outcome);
997
998        Ok(outcome)
999    }

1001    /// Returns the flattened [`FilterArgs`] arguments merged with [`Config`].
1002    /// Loads and applies filter from file if only last test run failures performed.
1003    pub fn filter(&self, config: &Config) -> Result<ProjectPathsAwareFilter> {
1004        let mut filter = self.filter.clone();
1005        if self.rerun {
1006            filter.test_pattern = last_run_failures(config);
1007        }
1008        if filter.path_pattern.is_some() {
1009            if self.path.is_some() {
1010                bail!("Can not supply both --match-path and |path|");
1011            }
1012        } else {
1013            filter.path_pattern = self.path.clone();
1014        }
1015        Ok(filter.merge_with_config(config))
1016    }

1073/// Lists all matching tests
1074fn list(runner: MultiContractRunner, filter: &ProjectPathsAwareFilter) -> Result<TestOutcome> {

1076
1077    if shell::is_json() {

1079    } else {
1080        for (file, contracts) in &results {

1091/// Load persisted filter (with last test run failures) from file.
1092fn last_run_failures(config: &Config) -> Option<regex::Regex> {
1093    match fs::read_to_string(&config.test_failures_file) {
1094        Ok(filter) => Some(Regex::new(&filter).unwrap()),
1095        Err(_) => None,
1096    }

1099/// Persist filter with last test run failures (only if there's any failure).
1100fn persist_run_failures(config: &Config, outcome: &TestOutcome) {
1101    if outcome.failed() > 0 && fs::create_file(&config.test_failures_file).is_ok() {
1102        let mut filter = String::new();
1103        let mut failures = outcome.failures().peekable();
1104        while let Some((test_name, _)) = failures.next() {
1105            if test_name.is_any_test()
1106                && let Some(test_match) = test_name.split("(").next()
1107            {
1108                filter.push_str(test_match);
1109                if failures.peek().is_some() {
1110                    filter.push('|');
1111                }
1112            }
1113        }
1114        let _ = fs::write(&config.test_failures_file, filter);
1115    }
1116}

1118/// Generate test report in JUnit XML report format.
1119fn junit_xml_report(results: &BTreeMap<String, SuiteResult>, verbosity: u8) -> Report {
1120    let mut total_duration = Duration::default();
1121    let mut junit_report = Report::new("Test run");
1122    junit_report.set_timestamp(Utc::now());
1123    for (suite_name, suite_result) in results {
1124        let mut test_suite = TestSuite::new(suite_name);
1125        total_duration += suite_result.duration;
1126        test_suite.set_time(suite_result.duration);
1127        test_suite.set_system_out(suite_result.summary());
1128        for (test_name, test_result) in &suite_result.test_results {
1129            let mut test_status = match test_result.status {
1130                TestStatus::Success => TestCaseStatus::success(),
1131                TestStatus::Failure => TestCaseStatus::non_success(NonSuccessKind::Failure),
1132                TestStatus::Skipped => TestCaseStatus::skipped(),
1133            };
1134            if let Some(reason) = &test_result.reason {
1135                test_status.set_message(reason);
1136            }
1137
1138            let mut test_case = TestCase::new(test_name, test_status);
1139            test_case.set_time(test_result.duration);
1140
1141            let mut sys_out = String::new();
1142            let result_report = test_result.kind.report();
1143            write!(sys_out, "{test_result} {test_name} {result_report}").unwrap();
1144            if verbosity >= 2 && !test_result.logs.is_empty() {
1145                write!(sys_out, "\\nLogs:\\n").unwrap();
1146                let console_logs = decode_console_logs(&test_result.logs);
1147                for log in console_logs {
1148                    write!(sys_out, "  {log}\\n").unwrap();
1149                }
1150            }
1151
1152            test_case.set_system_out(sys_out);
1153            test_suite.add_test_case(test_case);
1154        }
1155        junit_report.add_test_suite(test_suite);
1156    }
1157    junit_report.set_time(total_duration);
1158    junit_report
1159}
