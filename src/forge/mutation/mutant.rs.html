8/// Wraps an unary operator mutated, to easily store pre/post-fix op swaps
9#[derive(Debug, Clone)]
10pub struct UnaryOpMutated {
11    /// String containing the whole new expression (operator and its target)
12    /// eg `a++`
13    new_expression: String,
14
15    /// The underlying operator used by this mutant
16    pub resulting_op_kind: UnOpKind,
17}
18
19impl UnaryOpMutated {
20    pub fn new(new_expression: String, resulting_op_kind: UnOpKind) -> Self {
21        Self { new_expression, resulting_op_kind }
22    }
23}
24
25impl Display for UnaryOpMutated {
26    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
27        write!(f, "{}", self.new_expression)
28    }
29}

33#[derive(Debug, Clone)]
34pub enum MutationType {
35    // @todo Solar doesn't differentiate numeric type in LitKind (only on declaration?) -> for
36    // now, planket and let solc filter out the invalid mutants -> we might/should add a
37    // hashtable of the var to their underlying type (signed or not) so we avoid *a lot* of
38    // invalid mutants
39    /// For an initializer x, of type
40    /// bool: replace x with !x
41    /// uint: replace x with 0
42    /// int: replace x with 0; replace x with -x (temp: this is mutated for uint as well)
43    ///
44    /// For a binary op y: apply BinaryOp(y)
45    Assignment(AssignVarTypes),
46
47    /// For a binary op y in BinOpKind ("+", "-", ">=", etc)
48    /// replace y with each non-y in op
49    BinaryOp(BinOpKind),
50
51    /// For a delete expr x `delete foo`, replace x with `assert(true)`
52    DeleteExpression,
53
54    /// replace "delegatecall" with "call"
55    ElimDelegate,
56
57    /// Gambit doesn't implement nor define it?
58    FunctionCall,
59
60    // /// For a if(x) condition x:
61    // /// replace x with true; replace x with false
62    // This mutation is not used anymore, as we mutate the condition as an expression,
63    // which will creates true/false mutant as well as more complex conditions (eg if(foo++ >
64    // --bar) ) IfStatementMutation,
65    /// For a require(x) condition:
66    /// replace x with true; replace x with false
67    // Same as for IfStatementMutation, the expression inside the require is mutated as an
68    // expression to handle increment etc
69    Require,
70
71    // @todo review if needed -> this might creates *a lot* of combinations for super-polyadic fn
72    // tho       only swapping same type (to avoid obvious compilation failure), but should
73    // take into account       implicit casting too...
74    /// For 2 args of the same type x,y in a function args:
75    /// swap(x, y)
76    SwapArgumentsFunction,
77
78    // @todo same remark as above, might end up in a space too big to explore + filtering out
79    // based on type
80    /// For an expr taking 2 expression x, y (x+y, x-y, x = x + ...):
81    /// swap(x, y)
82    SwapArgumentsOperator,
83
84    /// For an unary operator x in UnOpKind (eg "++", "--", "~", "!"):
85    /// replace x with all other operator in op
86    /// Pre or post- are different UnOp
87    UnaryOperator(UnaryOpMutated),
88}

90impl MutationType {
91    fn get_name(&self) -> String {
92        match self {
93            Self::Assignment(var_type) => match var_type {
94                AssignVarTypes::Literal(kind) => {
95                    format!("{}_{}", "Assignment", kind.description())
96                }
97                AssignVarTypes::Identifier(ident) => {
98                    format!("{}_{}", "Assignment", ident)
99                }
100            },
101            Self::BinaryOp(kind) => {
102                format!("{}_{:?}", "BinaryOp", kind)
103            }
104            Self::DeleteExpression => "DeleteExpression".to_string(),
105            Self::ElimDelegate => "ElimDelegate".to_string(),
106            Self::FunctionCall => "FunctionCall".to_string(),
107            Self::Require => "Require".to_string(),
108            Self::SwapArgumentsFunction => "SwapArgumentsFunction".to_string(),
109            Self::SwapArgumentsOperator => "SwapArgumentsOperator".to_string(),
110            Self::UnaryOperator(mutated) => {
111                // avoid operator in tmp dir name
112                format!("{}_{:?}", "UnaryOperator", mutated.resulting_op_kind)
113            }
114        }
115    }
116}
117
118impl Display for MutationType {
119    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
120        match self {
121            Self::Assignment(kind) => match kind {
122                AssignVarTypes::Literal(kind) => match kind {
123                    LitKind::Number(val) => write!(f, "{val}"),
124                    LitKind::Bool(val) => write!(f, "{val}"),
125                    LitKind::Address(val) => write!(f, "{val}"),
126                    // Reachable?
127                    LitKind::Rational(val) => write!(f, "{val}"),
128                    LitKind::Str(_, val, _) => write!(f, "{val:?}"),
129                    LitKind::Err(val) => todo!(),
130                },
131                AssignVarTypes::Identifier(ident) => write!(f, "{ident}"),
132            },
133            Self::BinaryOp(kind) => write!(f, "{}", kind.to_str()),
134            Self::DeleteExpression => write!(f, "assert(true)"),
135            Self::ElimDelegate => write!(f, "call"),
136            Self::UnaryOperator(mutated) => write!(f, "{mutated}"),
137
138            Self::FunctionCall
139            | Self::Require
140            | Self::SwapArgumentsFunction
141            | Self::SwapArgumentsOperator => write!(f, ""),
142        }
143    }
144}

153/// A given mutation
154#[derive(Debug, Clone)]
155pub struct Mutant {
156    /// The path to the project root where this mutant (tries to) live
157    pub path: PathBuf,
158    pub span: Span,
159    pub mutation: MutationType,
160}
161
162impl Display for Mutant {
163    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
164        write!(
165            f,
166            "{}:{}-{}:{}",
167            self.path.display(),
168            self.span.lo().0,
169            self.span.hi().0,
170            self.mutation
171        )
172    }
173}

180    #[test]
181    fn test_mutation_type_get_name() {
182        assert_eq!(MutationType::DeleteExpression.get_name(), "DeleteExpression");
183        assert_eq!(MutationType::ElimDelegate.get_name(), "ElimDelegate");
184        assert_eq!(MutationType::FunctionCall.get_name(), "FunctionCall");
185        assert_eq!(MutationType::Require.get_name(), "Require");
186        assert_eq!(MutationType::SwapArgumentsFunction.get_name(), "SwapArgumentsFunction");
187        assert_eq!(MutationType::SwapArgumentsOperator.get_name(), "SwapArgumentsOperator");
188
189        assert_eq!(MutationType::BinaryOp(BinOpKind::Add).get_name(), "BinaryOp_Add");
190
191        let lit_num = LitKind::Number(123.into());
192        assert_eq!(
193            MutationType::Assignment(AssignVarTypes::Literal(lit_num)).get_name(),
194            "Assignment_number"
195        );
196
197        let ident = AssignVarTypes::Identifier("myVar".to_string());
198        assert_eq!(MutationType::Assignment(ident).get_name(), "Assignment_myVar");
199
200        let unary_mutated = UnaryOpMutated::new("a--".to_string(), UnOpKind::PreInc);
201        assert_eq!(MutationType::UnaryOperator(unary_mutated).get_name(), "UnaryOperator_PreInc");
202    }
