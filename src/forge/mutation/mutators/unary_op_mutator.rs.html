1use super::{MutationContext, Mutator};
2use crate::mutation::mutant::{Mutant, MutationType, UnaryOpMutated};
3use eyre::Result;
4use solar_parse::ast::{ExprKind, LitKind, UnOpKind};
5use std::path::PathBuf;
6
7pub struct UnaryOperatorMutator;
8
9impl Mutator for UnaryOperatorMutator {
10    fn generate_mutants(&self, context: &MutationContext<'_>) -> Result<Vec<Mutant>> {
11        let operations = vec![
12            UnOpKind::PreInc, // number
13            UnOpKind::PreDec, // n
14            UnOpKind::Neg,    // n @todo filter this one only for int
15            UnOpKind::BitNot, // n
16        ];
17
18        let post_fixed_operations = vec![UnOpKind::PostInc, UnOpKind::PostDec];
19
20        let expr = context.expr.unwrap();
21
22        let target_kind;
23        let op;
24
25        match &expr.kind {
26            ExprKind::Unary(un_op, target) => {
27                target_kind = &target.kind;
28                op = un_op.kind;
29            }
30            _ => unreachable!(),
31        };
32
33        let target_content = match target_kind {
34            ExprKind::Lit(lit, _) => match &lit.kind {
35                LitKind::Bool(val) => val.to_string(),
36                LitKind::Number(val) => val.to_string(),
37                _ => String::new(),
38            },
39            ExprKind::Ident(inner) => inner.to_string(),
40            ExprKind::Member(expr, ident) => {
41                match expr.kind {
42                    ExprKind::Ident(inner) => {
43                        format!("{}{}", ident.as_str(), inner.to_string())
44                    } // @todo not supporting something like a.b[0]++
45                    _ => String::new(),
46                }
47            }
48            _ => String::new(),
49        };
50
51        // Bool has only the Not operator as possible target -> we try removing it
52        if op == UnOpKind::Not {
53            return Ok(vec![Mutant {
54                span: expr.span,
55                mutation: MutationType::UnaryOperator(UnaryOpMutated::new(
56                    target_content.to_string(),
57                    UnOpKind::Not,
58                )),
59                path: context.path.clone(),
60            }]);
61        }
62
63        let mut mutations: Vec<Mutant>;
64
65        mutations = operations
66            .into_iter()
67            .filter(|&kind| kind != op)
68            .map(|kind| {
69                let new_expression = format!("{}{}", kind.to_str(), target_content);
70
71                let mutated = UnaryOpMutated::new(new_expression, kind);
72
73                Mutant {
74                    span: expr.span,
75                    mutation: MutationType::UnaryOperator(mutated),
76                    path: context.path.clone(),
77                }
78            })
79            .collect();
80
81        mutations.extend(post_fixed_operations.into_iter().filter(|&kind| kind != op).map(
82            |kind| {
83                let new_expression = format!("{}{}", target_content, kind.to_str());
84
85                let mutated = UnaryOpMutated::new(new_expression, kind);
86
87                Mutant {
88                    span: expr.span,
89                    mutation: MutationType::UnaryOperator(mutated),
90                    path: context.path.clone(),
91                }
92            },
93        ));
94
95        Ok(mutations)
96    }
97
98    fn is_applicable(&self, ctxt: &MutationContext<'_>) -> bool {
99        if let Some(expr) = ctxt.expr {
100            if let ExprKind::Unary(_, _) = &expr.kind {
101                return true;
102            }
103        }
104
105        false
106    }
107}
