1use super::{MutationContext, Mutator};
2use crate::mutation::mutant::{Mutant, MutationType};
3
4use eyre::Result;
5use solar_parse::ast::ExprKind;
6use std::{fmt::Display, path::PathBuf};
7
8pub struct ElimDelegateMutator;
9
10impl Mutator for ElimDelegateMutator {
11    fn generate_mutants(&self, context: &MutationContext<'_>) -> Result<Vec<Mutant>> {
12        Ok(vec![Mutant {
13            span: context.span,
14            mutation: MutationType::ElimDelegate,
15            path: context.path.clone(),
16        }])
17    }
18
19    fn is_applicable(&self, ctxt: &MutationContext<'_>) -> bool {
20        ctxt.expr
21            .as_ref()
22            .and_then(|expr| match &expr.kind {
23                ExprKind::Call(callee, _) => Some(callee),
24                _ => None,
25            })
26            .and_then(|callee| match &callee.kind {
27                ExprKind::Member(_, ident) => Some(ident),
28                _ => None,
29            })
30            .is_some_and(|ident| ident.to_string() == "delegatecall")
31    }
32}
33
34impl Display for ElimDelegateMutator {
35    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
36        write!(f, "")
37    }
38}
