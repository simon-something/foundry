9/// Registry of all available mutators (ie implementing the Mutator trait)
10pub struct MutatorRegistry {
11    mutators: Vec<Box<dyn Mutator>>,
12}
13
14impl MutatorRegistry {
15    pub fn default() -> Self {
16        let mut registry = Self { mutators: Vec::new() };
17
18        registry.mutators.push(Box::new(assignement_mutator::AssignmentMutator));
19        registry.mutators.push(Box::new(binary_op_mutator::BinaryOpMutator));
20        registry.mutators.push(Box::new(delete_expression_mutator::DeleteExpressionMutator));
21        registry.mutators.push(Box::new(elim_delegate_mutator::ElimDelegateMutator));
22        registry.mutators.push(Box::new(unary_op_mutator::UnaryOperatorMutator));
23
24        registry
25    }
26
27    pub fn new_with_mutators(mutators: Vec<Box<dyn Mutator>>) -> Self {
28        Self { mutators }
29    }
30
31    /// Find all applicable mutators for a given context and return the corresponding mutations
32    pub fn generate_mutations(&self, context: &MutationContext<'_>) -> Vec<Mutant> {
33        self.mutators
34            .iter()
35            .filter(|mutator| mutator.is_applicable(context))
36            .filter_map(|mutator| mutator.generate_mutants(context).ok())
37            .flatten()
38            .collect()
39    }
40}
