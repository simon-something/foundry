1use super::{MutationContext, Mutator};
2use crate::mutation::mutant::{Mutant, MutationType};
3use eyre::{OptionExt, Result};
4use solar_parse::ast::{BinOp, BinOpKind, ExprKind};
5use std::path::PathBuf;
6
7pub struct BinaryOpMutator;
8
9// @todo Add the other way to get there
10
11impl Mutator for BinaryOpMutator {
12    fn generate_mutants(&self, context: &MutationContext<'_>) -> Result<Vec<Mutant>> {
13        let bin_op = get_bin_op(context)?;
14        let op = bin_op.kind;
15
16        let operations_bools = vec![
17            // Bool
18            BinOpKind::Lt,
19            BinOpKind::Le,
20            BinOpKind::Gt,
21            BinOpKind::Ge,
22            BinOpKind::Eq,
23            BinOpKind::Ne,
24            BinOpKind::Or,
25            BinOpKind::And,
26        ]; // this cover the "if" mutations, as every other mutant is tested, at least once
27        // @todo to optimize -> replace whole stmt (need new visitor override for visit_stmt tho)
28        // with true/false and skip operations_bools here (mayve some "level"/depth of
29        // mutation as param?)
30
31        let operations_num_bitwise = vec![
32            // Arithm
33            BinOpKind::Shr,
34            BinOpKind::Shl,
35            BinOpKind::Sar,
36            BinOpKind::BitAnd,
37            BinOpKind::BitOr,
38            BinOpKind::BitXor,
39            BinOpKind::Add,
40            BinOpKind::Sub,
41            BinOpKind::Pow,
42            BinOpKind::Mul,
43            BinOpKind::Div,
44            BinOpKind::Rem,
45        ];
46
47        let operations =
48            if operations_bools.contains(&op) { operations_bools } else { operations_num_bitwise };
49
50        Ok(operations
51            .into_iter()
52            .filter(|&kind| kind != op)
53            .map(|kind| Mutant {
54                span: context.span,
55                mutation: MutationType::BinaryOp(kind),
56                path: context.path.clone(),
57            })
58            .collect())
59    }
60
61    fn is_applicable(&self, ctxt: &MutationContext<'_>) -> bool {
62        if ctxt.expr.is_none() {
63            return false;
64        }
65
66        match ctxt.expr.unwrap().kind {
67            ExprKind::Binary(_, _, _) => true,
68            ExprKind::Assign(_, bin_op, _) => bin_op.is_some(),
69            _ => false,
70        }
71    }
72}
73
74fn get_bin_op(ctxt: &MutationContext<'_>) -> Result<BinOp> {
75    let expr = ctxt.expr.ok_or_eyre("BinaryOpMutator: unexpected expression")?;
76
77    match expr.kind {
78        ExprKind::Assign(_, Some(bin_op), _) => Ok(bin_op),
79        ExprKind::Binary(_, op, _) => Ok(op),
80        _ => eyre::bail!("BinaryOpMutator: unexpected expression kind"),
81    }
82}
